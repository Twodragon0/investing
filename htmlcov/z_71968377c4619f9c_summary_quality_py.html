1"""Unified summary quality detectors — single source of truth. 

2 

3Public API: 

4 - ``ARTICLE_SPECIFIC_RE`` — canonical positive-signal pattern. Both 

5 ``common.enrichment`` and ``scripts.fix_post_descriptions`` import it 

6 from here, so a pattern update lands in one place. 

7 - ``GENERIC_DESC_PATTERNS`` — canonical list of generic/synthetic 

8 description regexes. ``common.summarizer`` consumes via the facade 

9 so the patterns live in exactly one location. 

10 - ``has_positive_signal(body)`` — True if body carries a real-content 

11 token (number, currency, acronym, headline lead-in). 

12 - ``is_generic_desc(desc)`` — True if desc matches any generic / 

13 synthetic placeholder pattern. Single SSoT for generic-desc detection. 

14 - ``is_boilerplate(desc)`` — orchestrates the three boilerplate detectors 

15 (``common.enrichment._is_site_boilerplate``, 

16 ``is_generic_desc`` defined here, 

17 ``common.summarizer._is_boilerplate_desc``). 

18 

19Circular-import resolution: this module is imported both as a consumer 

20(needs detectors from enrichment/summarizer) and as a provider (enrichment 

21needs ARTICLE_SPECIFIC_RE; summarizer needs GENERIC_DESC_PATTERNS). We bind 

22sibling modules with ``from . import <name> as _<name>_mod`` so each side 

23only resolves the other's attributes lazily inside function bodies. 

24Canonical patterns are defined BEFORE the sibling imports so partial-module 

25access from enrichment/summarizer during their own load finds the patterns. 

26 

27Lookaround over ``\\b``: in Python 3 ``\\w`` includes Hangul, so 

28``\\b\\d{4}\\b`` does NOT match ``2026년`` (no boundary between ``6`` and 

29``년``). ``(?<!\\d)`` / ``(?!\\d)`` is used instead. 

30"""

45# --------------------------------------------------------------------------- 

46# Canonical positive-signal pattern. 

47# Defined BEFORE sibling imports so enrichment.py can read it from a partial 

48# summary_quality module during its own load (circular import resolution). 

49# --------------------------------------------------------------------------- 

50ARTICLE_SPECIFIC_RE = re.compile( 

51 r"(?:[A-Z][a-z]+(?:\s+[A-Z][a-z]+)+)" # Title-case word pair (proper noun) 

52 r"|(?:(?<![A-Za-z])[A-Z]{2,}(?![A-Za-z]))" # Acronym / ticker 

53 r"|(?:(?<!\d)\d{4}(?!\d))" # 4-digit year (2026…) 

54 r"|(?:\d+[.,]\d+)" # Decimal/comma number 

55 r"|(?:[$€£₩¥]\s*\d)" # Currency + digit 

56 r"|(?:\d+\s*(?:%|억|만|조|달러|원|위안|위|건|종|개|월|일))" # Number with unit 

57 r"|(?:(?:월|년|일)\s*\d)" # Korean date fragment (월 04, 년 2026) 

58 r"|(?:\d{1,3}(?:,\d{3})+)" # 1,234,567 grouping 

59 r'|(?:["“‘][^"”’]{3,}["”’])' # Quoted phrase 

60 r"|(?:주요\s*(?:테마|출처|이벤트|이슈|종목)\s*:)" # Korean headline lead-in 

61 r"|(?:오늘의?\s*헤드라인\s*:)" # "오늘의 헤드라인:" 

62)

65def has_positive_signal(body: str) -> bool: 

66 """Return True if body carries at least one positive content signal.""" 

67 if not body: 

68 return False 

69 return bool(ARTICLE_SPECIFIC_RE.search(body))

130def is_generic_desc(desc: str) -> bool: 

131 """Return True if desc matches any generic/synthetic placeholder pattern. 

132 

133 Single SSoT consumed by ``common.summarizer._is_generic_desc`` (kept as a 

134 thin backward-compat wrapper) and by ``is_boilerplate`` orchestrator. 

135 """ 

136 if not desc: 

137 return False 

138 body = desc.strip() 

139 return any(p.search(body) for p in GENERIC_DESC_PATTERNS)

151def is_boilerplate(desc: str) -> bool: 

152 """Return True if desc matches any boilerplate / generic detector. 

153 

154 Orchestrates the three canonical detectors: site-level boilerplate 

155 (``common.enrichment._is_site_boilerplate``), generic-desc patterns 

156 (``is_generic_desc`` defined above), and the phrase-list filter 

157 (``common.summarizer._is_boilerplate_desc``). 

158 """ 

159 if not desc: 

160 return False 

161 if _enrichment_mod._is_site_boilerplate(desc): 

162 return True 

163 if is_generic_desc(desc): 

164 return True 

165 if _summarizer_mod._is_boilerplate_desc(desc): 

166 return True 

167 return False
