1"""Unified summary quality detectors — single source of truth. 

2 

3Public API: 

4 - ``ARTICLE_SPECIFIC_RE`` — canonical positive-signal pattern. Both 

5 ``common.enrichment`` and ``scripts.fix_post_descriptions`` import it 

6 from here, so a pattern update lands in one place. 

7 - ``is_boilerplate(desc)`` — orchestrates the three boilerplate detectors 

8 (``common.enrichment._is_site_boilerplate``, 

9 ``common.summarizer._is_generic_desc``, 

10 ``common.summarizer._is_boilerplate_desc``). 

11 - ``has_positive_signal(body)`` — True if body carries a real-content 

12 token (number, currency, acronym, headline lead-in). 

13 

14Circular-import resolution: this module is imported both as a consumer 

15(needs detectors from enrichment/summarizer) and as a provider (enrichment 

16needs ARTICLE_SPECIFIC_RE). We bind sibling modules with 

17``from . import <name> as _<name>_mod`` so each side only resolves the 

18other's attributes lazily inside function bodies. ARTICLE_SPECIFIC_RE is 

19defined BEFORE the sibling imports so partial-module access from 

20enrichment during its own load finds the pattern. 

21 

22Lookaround over ``\\b``: in Python 3 ``\\w`` includes Hangul, so 

23``\\b\\d{4}\\b`` does NOT match ``2026년`` (no boundary between ``6`` and 

24``년``). ``(?<!\\d)`` / ``(?!\\d)`` is used instead. 

25""" 

26 

27from __future__ import annotations 

28 

29import re 

30 

31__all__ = [ 

32 "ARTICLE_SPECIFIC_RE", 

33 "has_positive_signal", 

34 "is_boilerplate", 

35] 

36

38# --------------------------------------------------------------------------- 

39# Canonical positive-signal pattern. 

40# Defined BEFORE sibling imports so enrichment.py can read it from a partial 

41# summary_quality module during its own load (circular import resolution). 

42# --------------------------------------------------------------------------- 

43ARTICLE_SPECIFIC_RE = re.compile( 

44 r"(?:[A-Z][a-z]+(?:\s+[A-Z][a-z]+)+)" # Title-case word pair (proper noun) 

45 r"|(?:(?<![A-Za-z])[A-Z]{2,}(?![A-Za-z]))" # Acronym / ticker 

46 r"|(?:(?<!\d)\d{4}(?!\d))" # 4-digit year (2026…) 

47 r"|(?:\d+[.,]\d+)" # Decimal/comma number 

48 r"|(?:[$€£₩¥]\s*\d)" # Currency + digit 

49 r"|(?:\d+\s*(?:%|억|만|조|달러|원|위안|위|건|종|개|월|일))" # Number with unit 

50 r"|(?:(?:월|년|일)\s*\d)" # Korean date fragment (월 04, 년 2026) 

51 r"|(?:\d{1,3}(?:,\d{3})+)" # 1,234,567 grouping 

52 r'|(?:["“‘][^"”’]{3,}["”’])' # Quoted phrase 

53 r"|(?:주요\s*(?:테마|출처|이벤트|이슈|종목)\s*:)" # Korean headline lead-in 

54 r"|(?:오늘의?\s*헤드라인\s*:)" # "오늘의 헤드라인:" 

55) 

56 

57 

58def has_positive_signal(body: str) -> bool: 

59 """Return True if body carries at least one positive content signal.""" 

60 if not body: 

61 return False 

62 return bool(ARTICLE_SPECIFIC_RE.search(body)) 

63

68# because attribute lookup happens at call time, not load time. 

69# --------------------------------------------------------------------------- 

70from . import enrichment as _enrichment_mod # noqa: E402 

71from . import summarizer as _summarizer_mod # noqa: E402

74def is_boilerplate(desc: str) -> bool: 

75 """Return True if desc matches any boilerplate / generic detector. 

76 

77 Orchestrates the three canonical detectors in ``common.enrichment`` / 

78 ``common.summarizer``. This function is the single dependency that 

79 consumers should import. 

80 """ 

81 if not desc: 

82 return False 

83 if _enrichment_mod._is_site_boilerplate(desc): 

84 return True 

85 if _summarizer_mod._is_generic_desc(desc): 

86 return True 

87 if _summarizer_mod._is_boilerplate_desc(desc): 

88 return True 

89 return False
