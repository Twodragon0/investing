164 """Parse front matter and body from a post file.

166 Returns (front_matter_dict, body_string). 

167 Front matter values are kept as raw strings. 

168 """ 

169 if not content.startswith("---"): 

170 return {}, content 

171 

172 end = content.index("---", 3) 

173 fm_raw = content[3:end].strip() 

174 body = content[end + 3 :].lstrip("\n") 

175 

176 fm: dict[str, str] = {} 

177 for line in fm_raw.split("\n"): 

178 if ":" in line: 

179 key, _, val = line.partition(":") 

180 fm[key.strip()] = val.strip()

198def clean_description(fm: dict[str, str]) -> bool: 

199 """Clean description: remove HTML, markdown blockquote, trim to 160 chars.""" 

200 desc = fm.get("description", "") 

201 if not desc: 

202 return False 

203 

204 original = desc 

205 # Strip surrounding quotes 

206 if (desc.startswith('"') and desc.endswith('"')) or (desc.startswith("'") and desc.endswith("'")): 

207 inner = desc[1:-1] 

208 else: 

209 inner = desc 

210 

211 # Remove HTML tags 

212 inner = re.sub(r"<[^>]+>", "", inner) 

213 # Remove leading > (blockquote) 

214 inner = re.sub(r"^>\s*", "", inner) 

215 # Remove markdown bold 

216 inner = re.sub(r"\*\*([^*]+)\*\*", r"\1", inner) 

217 # Remove trailing raw tag strings (e.g., "주요 키워드: crypto, news, daily-digest.") 

218 inner = re.sub(r"\s*주요 키워드:[\s\w,.-]+\.?\s*$", "", inner) 

219 # Collapse whitespace 

220 inner = re.sub(r"\s+", " ", inner).strip() 

221 

222 # Truncate to 160 chars at a word boundary 

223 if len(inner) > 160: 

224 truncated = inner[:157] 

225 last_space = truncated.rfind(" ") 

226 if last_space > 100: 

227 truncated = truncated[:last_space] 

228 inner = truncated.rstrip(".,;:") + "..." 

229 

230 new_desc = f'"{inner}"' 

231 if new_desc != original: 

232 fm["description"] = new_desc 

233 return True 

234 return False

320def remove_intro_duplication_in_summary(body: str) -> tuple[str, bool]: 

321 """Remove lines in '전체 뉴스 요약' that duplicate the intro paragraph.""" 

322 # Extract intro: first non-empty paragraph BEFORE any ## heading 

323 first_heading = re.search(r"^## ", body, re.MULTILINE) 

324 intro_region = body[: first_heading.start()] if first_heading else "" 

325 

326 intro_text = "" 

327 for line in intro_region.split("\n"): 

328 stripped = line.strip() 

329 if stripped and not stripped.startswith("<") and not stripped.startswith("---"): 

330 # Remove markdown bold for comparison 

331 intro_text = re.sub(r"\*\*([^*]+)\*\*", r"\1", stripped) 

332 break 

333 

334 if not intro_text or len(intro_text) < 20: 

335 return body, False 

336 

337 # Find the summary section 

338 summary_pattern = re.compile(r"^## 전체 뉴스 요약\s*$", re.MULTILINE) 

339 match = summary_pattern.search(body) 

340 if not match: 

341 return body, False 

342 

343 # Find the end of the summary section (next ## heading or end) 

344 section_start = match.end() 

345 next_heading = re.search(r"^## ", body[section_start:], re.MULTILINE) 

346 if next_heading: 

347 section_end = section_start + next_heading.start() 

348 else: 

349 section_end = len(body) 

350 

351 section = body[section_start:section_end] 

352 new_lines = [] 

353 changed = False 

354 

355 # Normalize intro for fuzzy matching 

356 intro_norm = re.sub(r"[*\s]+", " ", intro_text).strip().lower() 

357 

358 for line in section.split("\n"): 

359 line_stripped = line.strip() 

360 if line_stripped.startswith("- ") or line_stripped.startswith("* "): 

361 bullet_content = line_stripped[2:].strip() 

362 else: 

363 bullet_content = line_stripped 

364 

365 # Normalize for comparison 

366 bullet_norm = re.sub(r"[*\s]+", " ", bullet_content).strip().lower() 

367 

368 # Check if this line is substantially similar to intro 

369 if bullet_norm and intro_norm and len(bullet_norm) > 20: 

370 # Check if intro is contained in bullet or vice versa 

371 if bullet_norm in intro_norm or intro_norm in bullet_norm: 

372 changed = True 

373 continue 

374 # Check overlap ratio 

375 intro_words = set(intro_norm.split()) 

376 bullet_words = set(bullet_norm.split()) 

377 if intro_words and bullet_words: 

378 overlap = len(intro_words & bullet_words) / max(len(intro_words), len(bullet_words)) 

379 if overlap > 0.7: 

380 changed = True 

381 continue 

382 

383 new_lines.append(line) 

384 

385 if changed: 

386 new_section = "\n".join(new_lines) 

387 body = body[:section_start] + new_section + body[section_end:] 

388 

389 return body, changed

392def remove_summary_article_duplicates(body: str) -> tuple[str, bool]: 

393 """Remove summary list items that duplicate numbered article titles below. 

394 

395 Detects patterns like: 

396 - 1. Title text here. - Source Title text here. 

397 where the same title appears again in a numbered entry in a later section. 

398 Also removes list items with "- N." prefix that repeat themselves internally. 

399 """ 

400 original = body 

401 lines = body.split("\n") 

402 new_lines = [] 

403 # Collect all numbered article titles (e.g. "**1. [Title..." or "**1. Title...") 

404 article_titles = set() 

405 for line in lines: 

406 m = re.match(r"\*\*\d+\.\s*(?:\[)?(.{15,80}?)(?:\]|\*)", line) 

407 if m: 

408 article_titles.add(m.group(1).strip()[:50]) 

409 

410 for line in lines: 

411 # Check for "- 1. Title - Source Title" pattern with internal duplication 

412 m = re.match(r"^- \d+\.\s+(.+)$", line) 

413 if m: 

414 content = m.group(1) 

415 # Check if first half repeats in second half (internal dup) 

416 half = len(content) // 2 

417 if half > 20 and content[:half].strip().rstrip(".") in content[half:]: 

418 continue 

419 # Check if this title already appears as a numbered article 

420 clean = re.sub(r"\s*-\s*\S+\s*$", "", content).strip()[:50] 

421 if clean and any(clean in t or t in clean for t in article_titles): 

422 continue 

423 new_lines.append(line) 

424 

425 result = "\n".join(new_lines) 

426 return result, result != original

436 section_start = match.end() 

437 next_heading = re.search(r"^## ", body[section_start:], re.MULTILINE) 

438 if next_heading: 

439 section_end = section_start + next_heading.start() 

440 else: 

441 # Look for --- separator or end of file 

442 separator = re.search(r"^---\s*$", body[section_start:], re.MULTILINE) 

443 if separator: 

444 section_end = section_start + separator.start() 

445 else: 

446 section_end = len(body) 

447 

448 section = body[section_start:section_end] 

449 seen_articles: set[str] = set() 

450 new_lines: list[str] = [] 

451 changed = False 

452 

453 for line in section.split("\n"): 

454 # Detect "주요 기사" lines 

455 if "주요 기사:" in line or "주요 기사*:" in line: 

456 # Extract article title for dedup 

457 article_match = re.search(r"\*([^*]+)\*", line) 

458 if article_match: 

459 title = article_match.group(1).strip() 

460 if title in seen_articles: 

461 changed = True 

462 continue 

463 seen_articles.add(title) 

464 new_lines.append(line) 

465 

466 if changed: 

467 new_section = "\n".join(new_lines) 

468 body = body[:section_start] + new_section + body[section_end:] 

469 

470 return body, changed

473def _extract_themes_from_insight(line: str) -> tuple[str, str]: 

474 """Extract two theme names from the insight opening line.""" 

475 # Pattern: "오늘 가장 주목할 테마는 **비트코인**와 **가격/시장**입니다" 

476 # or with emoji: "**🟠 비트코인**(52건)과 **📈 가격/시장**(26건)" 

477 theme_matches = re.findall( 

478 r"\*\*(?:[^\*]*?)(비트코인|가격/시장|거래소|규제/정책|이더리움|" 

479 r"정치/정책|AI/기술|매크로/금리|DeFi|NFT|스테이블코인|알트코인)\*\*", 

480 line, 

481 ) 

482 if len(theme_matches) >= 2: 

483 return theme_matches[0], theme_matches[1] 

484 return "", ""

487def replace_generic_insight(body: str) -> tuple[str, bool]: 

488 """Replace generic insight phrases with theme-specific analysis.""" 

489 changed = False 

490 

491 for pattern in GENERIC_INSIGHT_PATTERNS: 

492 match = pattern.search(body) 

493 if not match: 

494 continue 

495 

496 # Find the context line to extract themes 

497 match_start = match.start() 

498 # Look backward for the line containing theme info 

499 preceding = body[:match_start] 

500 preceding_lines = preceding.split("\n") 

501 

502 theme1, theme2 = "", "" 

503 # Check the same line and previous lines for theme info 

504 for check_line in reversed(preceding_lines[-3:]): 

505 theme1, theme2 = _extract_themes_from_insight(check_line) 

506 if theme1 and theme2: 

507 break 

508 

509 # Also check the matched line itself 

510 if not (theme1 and theme2): 

511 line_start = body.rfind("\n", 0, match_start) + 1 

512 line_end = body.find("\n", match.end()) 

513 if line_end == -1: 

514 line_end = len(body) 

515 full_line = body[line_start:line_end] 

516 theme1, theme2 = _extract_themes_from_insight(full_line) 

517 

518 if theme1 and theme2: 

519 # Try both orderings 

520 replacement = THEME_INSIGHTS.get( 

521 (theme1, theme2), 

522 THEME_INSIGHTS.get((theme2, theme1), FALLBACK_INSIGHT), 

523 ) 

524 else: 

525 replacement = FALLBACK_INSIGHT 

526 

527 body = body[: match.start()] + replacement + body[match.end() :] 

528 changed = True 

529 break # Only one pattern per post expected 

530 

531 return body, changed

534def add_korean_to_keywords(body: str) -> tuple[str, bool]: 

535 """Add Korean translations to monitoring keywords. 

536 

537 Transform: **bitcoin**(37회) -> **bitcoin(비트코인)**(37회) 

538 Transform: **bitcoin** -> **bitcoin(비트코인)** 

539 """ 

540 changed = False 

541 

542 def _replace_keyword(m: re.Match[str]) -> str: 

543 nonlocal changed 

544 keyword = m.group(1) 

545 rest = m.group(2) or "" # e.g., "(37회)" or empty 

546 

547 # Skip if already has Korean in parens 

548 if re.search(r"\([가-힣]+\)", keyword): 

549 return m.group(0) 

550 

551 ko = KEYWORD_KO.get(keyword.lower()) 

552 if ko: 

553 changed = True 

554 return f"**{keyword}({ko})**{rest}" 

555 return m.group(0) 

556 

557 # Pattern: **keyword**(N회) or **keyword** 

558 body = re.sub( 

559 r"\*\*([a-zA-Z0-9/]+)\*\*(\(\d+회\))?", 

560 _replace_keyword, 

561 body, 

562 ) 

563 return body, changed

566def clean_keyword_none_artifacts(body: str) -> tuple[str, bool]: 

567 new_body = body 

568 new_body = re.sub(r"(\*\*[^*]+\*\*)None\(", r"\1(", new_body) 

569 new_body = re.sub(r"(\*\*[^*]+\*\*)None\s+\(", r"\1 (", new_body) 

570 new_body = re.sub(r"(\*\*[^*]+\*\*)None(?=[가-힣])", r"\1", new_body) 

571 new_body = re.sub(r"(\*\*[^*]+\*\*)None(?=\s|[.,:;!?])", r"\1", new_body) 

572 return new_body, new_body != body

575def clean_empty_data_sections(body: str) -> tuple[str, bool]: 

576 """Remove lines with 'data unavailable' messages and clean up empty sections.""" 

577 changed = False 

578 

579 empty_patterns = [ 

580 r"^>?\s*글로벌 암호화폐 시장 데이터를 일시적으로 가져올 수 없습니다\.?\s*$", 

581 r"^>?\s*코인 데이터를 일시적으로 가져올 수 없습니다\.?\s*$", 

582 r"^>?\s*\*?트렌딩 데이터를 가져올 수 없습니다\.?\*?\s*$", 

583 r"^>?\s*\*?데이터를 가져올 수 없습니다\.?\*?\s*$", 

584 r"^>?\s*매크로 경제 지표를 일시적으로 가져올 수 없습니다.*$", 

585 r"^>?\s*\*?데이터 없음\*?\s*$", 

586 r"^-?\s*\*?Alpha Vantage.*데이터를 가져올 수 없습니다.*\*?\s*$", 

587 r"^>?\s*\*?고래 거래 데이터를 가져올 수 없습니다\.?\*?\s*$", 

588 r"^>?\s*\*?Alpha Vantage API를 통한.*데이터를 가져올 수 없습니다.*\*?\s*$", 

589 ] 

590 

591 for pat_str in empty_patterns: 

592 pat = re.compile(pat_str, re.MULTILINE) 

593 if pat.search(body): 

594 body = pat.sub("", body) 

595 changed = True 

596 

597 # Remove table rows containing "데이터 없음" 

598 table_row_pat = re.compile(r"^\|[^|]*\|[^|]*\|\s*데이터 없음\s*\|\s*$", re.MULTILINE) 

599 if table_row_pat.search(body): 

600 body = table_row_pat.sub("", body) 

601 changed = True 

602 

603 return body, changed

612def remove_duplicate_articles_in_themes(body: str) -> tuple[str, bool]: 

613 """Remove duplicate articles across theme sections. 

614 

615 When the same article (by URL) appears as #1 in multiple theme sections, 

616 the duplicates are flagged. We don't remove the entire entry because 

617 each theme section should list its articles, but we remove exact 

618 duplicate italic summary lines above numbered entries. 

619 """ 

620 changed = False 

621 

622 # Find theme sections: ### emoji theme (Nk) 

623 theme_pattern = re.compile(r"^### .+? \(\d+건\)\s*$", re.MULTILINE) 

624 matches = list(theme_pattern.finditer(body)) 

625 

626 if len(matches) < 2: 

627 return body, changed 

628 

629 # For each theme section, find the italic summary line (standalone *text*) 

630 # and check if it duplicates a previous one 

631 seen_summaries: set[str] = set() 

632 replacements: list[tuple[int, int, str]] = [] 

633 

634 for i, m in enumerate(matches): 

635 section_start = m.end() 

636 if i + 1 < len(matches): 

637 section_end = matches[i + 1].start() 

638 else: 

639 # Find next ## heading or --- 

640 next_h2 = re.search(r"^## ", body[section_start:], re.MULTILINE) 

641 next_sep = re.search(r"^---\s*$", body[section_start:], re.MULTILINE) 

642 ends = [] 

643 if next_h2: 

644 ends.append(section_start + next_h2.start()) 

645 if next_sep: 

646 ends.append(section_start + next_sep.start()) 

647 section_end = min(ends) if ends else len(body) 

648 

649 section = body[section_start:section_end] 

650 

651 # Find standalone italic line: starts with \n*text*\n 

652 italic_match = re.search(r"\n(\*[^*\n]+\*)\s*\n", section) 

653 if italic_match: 

654 summary_text = italic_match.group(1).strip() 

655 if summary_text in seen_summaries: 

656 # Remove this duplicate italic line 

657 abs_start = section_start + italic_match.start() 

658 abs_end = section_start + italic_match.end() 

659 replacements.append((abs_start, abs_end, "\n")) 

660 changed = True 

661 else: 

662 seen_summaries.add(summary_text) 

663 

664 # Apply replacements in reverse order 

665 for start, end, replacement in reversed(replacements): 

666 body = body[:start] + replacement + body[end:] 

667 

668 return body, changed

671def fix_translation_artifacts(body: str) -> tuple[str, bool]: 

672 """Fix known machine translation artifacts in post body. 

673 

674 Applies _MISTRANSLATION_FIXES from post_generator, particle corrections, 

675 name fixes, and awkward phrasing cleanup. 

676 """ 

677 original = body 

678 

679 # Apply centralized mistranslation dictionary first 

680 try: 

681 from common.post_generator import _MISTRANSLATION_FIXES 

682 

683 for wrong, correct in _MISTRANSLATION_FIXES.items(): 

684 body = body.replace(wrong, correct) 

685 except ImportError: 

686 pass 

687 

688 fixes = [ 

689 # Particle corrections (names without 받침) 

690 ("트럼프은 ", "트럼프는 "), 

691 ("트럼프을 ", "트럼프를 "),

694 ("테슬라이 ", "테슬라가 "),

696 ("엔비디아이 ", "엔비디아가 "), 

697 ("메타은 ", "메타는 "), 

698 ("메타이 ", "메타가 "), 

699 ("오바마은 ", "오바마는 "), 

700 ("오바마이 ", "오바마가 "),

703 ("시과의", "시진핑과의"), 

704 ("시과 ", "시진핑과 "), 

705 ] 

706 for wrong, correct in fixes: 

707 body = body.replace(wrong, correct)

711 body = re.sub(r"말했습니까\?", "밝혔나?", body) 

712 # Inline picture tag separation 

713 body = re.sub(r"([^\n>])<picture>", r"\1\n\n<picture>", body) 

714 # Double text bug 

715 body = re.sub(r"시장 영향 가능성이 있는성이 있는", "시장 영향 가능성이 있는", body) 

716 

717 return body, body != original

725def process_post(filepath: Path, dry_run: bool = False) -> dict[str, int]: 

726 """Process a single post file. Returns dict of improvement counts.""" 

727 content = filepath.read_text(encoding="utf-8") 

728 fm, body = parse_post(content) 

729 

730 if not fm: 

731 return {} 

732 

733 stats: dict[str, int] = {} 

734 

735 # 1. Clean description 

736 if clean_description(fm): 

737 stats["description_cleaned"] = 1 

738 

739 metadata_stats = rebuild_low_quality_metadata(fm, body) 

740 stats.update(metadata_stats) 

741 

742 # 2. Remove intro duplication in summary 

743 body, did_change = remove_intro_duplication_in_summary(body) 

744 if did_change: 

745 stats["intro_dedup"] = 1 

746 

747 body, did_change = remove_summary_article_duplicates(body) 

748 if did_change: 

749 stats["summary_article_dedup"] = 1 

750 

751 # 3. Remove duplicate articles in insight 

752 body, did_change = remove_duplicate_articles_in_insight(body) 

753 if did_change: 

754 stats["insight_article_dedup"] = 1 

755 

756 # 4. Replace generic insight phrases 

757 body, did_change = replace_generic_insight(body) 

758 if did_change: 

759 stats["insight_improved"] = 1 

760 

761 # 5. Add Korean to monitoring keywords 

762 body, did_change = add_korean_to_keywords(body) 

763 if did_change: 

764 stats["keywords_translated"] = 1 

765 

766 # 6. Clean empty data sections 

767 body, did_change = clean_empty_data_sections(body) 

768 if did_change: 

769 stats["empty_sections_cleaned"] = 1 

770 

771 body, did_change = clean_keyword_none_artifacts(body) 

772 if did_change: 

773 stats["keyword_none_cleaned"] = 1 

774 

775 body, did_change = remove_duplicate_articles_in_themes(body) 

776 if did_change: 

777 stats["theme_summary_dedup"] = 1 

778 

779 body, did_change = sync_summary_total_count(body) 

780 if did_change: 

781 stats["summary_total_synced"] = 1 

782 

783 body, did_change = fix_markdown_link_artifacts(body) 

784 if did_change: 

785 stats["markdown_fixed"] = 1 

786 

787 # Remove "주요 키워드: ..." tails from body text (HTML and markdown) 

788 _body_before = body 

789 body = re.sub(r"\.\s*주요 키워드:[^<\n]{3,80}\.", ".", body) 

790 body = re.sub(r"\s*주요 키워드:[^<\n]{3,80}\.?(?=</)", "", body) 

791 if body != _body_before: 

792 stats["body_keyword_tail_cleaned"] = 1 

793 

794 body, did_change = collapse_blank_lines(body) 

795 if did_change: 

796 stats["blank_lines_collapsed"] = 1 

797 

798 # 9. Fix translation artifacts (particles, names, phrasing) 

799 body, did_change = fix_translation_artifacts(body) 

800 if did_change: 

801 stats["translation_fixed"] = 1 

802 

803 if not stats: 

804 return {} 

805 

806 # Write back 

807 new_content = serialize_front_matter(fm) + "\n\n" + body 

808 if not dry_run: 

809 filepath.write_text(new_content, encoding="utf-8") 

810 

811 return stats
