127def _prune_old_entries(entries: List[Dict[str, Any]], max_age_days: int = _MAX_AGE_DAYS) -> List[Dict[str, Any]]: 

128 """max_age_days 이전 항목을 제거한다.""" 

129 cutoff = (datetime.now(UTC) - timedelta(days=max_age_days)).strftime("%Y-%m-%d") 

130 pruned = [e for e in entries if e.get("date", "") >= cutoff] 

131 removed = len(entries) - len(pruned) 

132 if removed: 

133 logger.info("signal_history: %d개 오래된 항목 제거 (cutoff=%s)", removed, cutoff) 

134 return pruned

158class SignalTracker: 

159 """일별 신호 결과 저장 및 정확도 추적 클래스.

169 def __init__(self, history_path: str = _HISTORY_FILE) -> None: 

170 """초기화. 

171 

172 Args: 

173 history_path: 히스토리 JSON 파일 경로. 기본값은 _state/signal_history.json.

175 self._path = history_path 

176 self._store = TimeSeriesStore(Path(history_path), _SIGNAL_SCHEMA, logger) 

177 self._entries: List[Dict[str, Any]] = self._store.load(validate=False) 

178 logger.debug("SignalTracker 초기화: %d개 항목 로드", len(self._entries))

182 def record( 

183 self, 

184 result: CompositeResult, 

185 btc_price: Optional[float] = None, 

186 date: Optional[str] = None, 

187 ) -> SignalSnapshot: 

188 """당일 신호 결과를 히스토리에 저장한다. 

189 

190 이전 예측(전날 항목)이 있으면 btc_price와 비교하여 정확도를 역산 기록한다. 

191 

192 Args: 

193 result: SignalComposer.compose_signals() 반환값. 

194 btc_price: 당일 BTC 현재 가격 (USD). 없으면 정확도 평가 생략. 

195 date: YYYY-MM-DD 형식 날짜. 없으면 KST 오늘 날짜. 

196 

197 Returns: 

198 저장된 SignalSnapshot 객체. 

199 """ 

200 if date is None: 

201 from .config import get_kst_now 

202 

203 date = get_kst_now().strftime("%Y-%m-%d") 

204 

205 now_utc = datetime.now(UTC).strftime("%Y-%m-%dT%H:%M:%SZ") 

206 

207 # 개별 신호 점수 추출 

208 signal_scores: Dict[str, float] = {sr.name: round(sr.normalized, 4) for sr in result.signal_results} 

209 

210 snapshot = SignalSnapshot( 

211 date=date, 

212 composite_score=result.score, 

213 verdict=result.verdict, 

214 confidence=result.confidence, 

215 signal_scores=signal_scores, 

216 btc_price=btc_price, 

217 recorded_at=now_utc, 

218 ) 

219 

220 # btc_price가 있으면 전날 예측의 정확도를 업데이트한다 

221 if btc_price is not None: 

222 self._update_previous_accuracy(date, btc_price) 

223 

224 # 기존 항목 교체 또는 추가 (같은 날짜가 이미 있으면 덮어쓴다) 

225 entry_dict = asdict(snapshot) 

226 existing_idx = next((i for i, e in enumerate(self._entries) if e.get("date") == date), None) 

227 if existing_idx is not None: 

228 logger.info("기존 항목 덮어쓰기 (date=%s)", date) 

229 self._entries[existing_idx] = entry_dict 

230 else: 

231 self._entries.append(entry_dict) 

232 

233 # 오래된 항목 정리 

234 self._entries = _prune_old_entries(self._entries) 

235 

236 # 원자적 저장 (TimeSeriesStore 위임) 

237 self._store._write_atomic(self._entries) 

238 logger.info("신호 기록 완료: date=%s score=%.1f verdict=%s", date, result.score, result.verdict) 

239 return snapshot

241 def accuracy_report(self, lookback_days: int = 30) -> AccuracyReport: 

242 """최근 N일 예측 정확도 리포트를 생성한다. 

243 

244 Args: 

245 lookback_days: 분석할 최근 일수 (기본 30일). 

246 

247 Returns: 

248 AccuracyReport: 승률, 버딕트별 통계, 최근 레코드 포함. 

249 """ 

250 cutoff = (datetime.now(UTC) - timedelta(days=lookback_days)).strftime("%Y-%m-%d") 

251 recent = [e for e in self._entries if e.get("date", "") >= cutoff] 

252 

253 records: List[AccuracyRecord] = [] 

254 for entry in recent: 

255 acc = entry.get("accuracy") 

256 if acc is None: 

257 continue 

258 records.append( 

259 AccuracyRecord( 

260 date=entry["date"], 

261 predicted_verdict=acc.get("predicted_verdict", ""), 

262 predicted_score=acc.get("predicted_score", 0.0), 

263 actual_price_change_pct=acc.get("actual_price_change_pct"), 

264 actual_direction=acc.get("actual_direction"), 

265 correct=acc.get("correct"), 

266 ) 

267 ) 

268 

269 evaluable = [r for r in records if r.correct is not None] 

270 correct_count = sum(1 for r in evaluable if r.correct) 

271 win_rate = correct_count / len(evaluable) if evaluable else 0.0 

272 

273 # 버딕트 유형별 집계 

274 by_verdict: Dict[str, Dict[str, Any]] = {} 

275 for r in evaluable: 

276 v = r.predicted_verdict 

277 if v not in by_verdict: 

278 by_verdict[v] = {"total": 0, "correct": 0, "win_rate": 0.0} 

279 by_verdict[v]["total"] += 1 

280 if r.correct: 

281 by_verdict[v]["correct"] += 1 

282 for _v, stats in by_verdict.items(): 

283 stats["win_rate"] = stats["correct"] / stats["total"] if stats["total"] else 0.0 

284 

285 return AccuracyReport( 

286 total_predictions=len(evaluable), 

287 correct_predictions=correct_count, 

288 win_rate=win_rate, 

289 by_verdict=by_verdict, 

290 recent_accuracy=sorted(records, key=lambda r: r.date, reverse=True)[:lookback_days], 

291 )

293 def get_history(self, days: int = 30) -> List[SignalSnapshot]: 

294 """최근 N일 스냅샷을 반환한다. 

295 

296 Args: 

297 days: 반환할 최근 일수. 

298 

299 Returns: 

300 날짜 역순(최신 → 오래된) SignalSnapshot 리스트. 

301 """ 

302 cutoff = (datetime.now(UTC) - timedelta(days=days)).strftime("%Y-%m-%d") 

303 snapshots = [] 

304 for e in self._entries: 

305 if e.get("date", "") >= cutoff: 

306 try: 

307 snapshots.append( 

308 SignalSnapshot( 

309 date=e["date"], 

310 composite_score=e.get("composite_score", 0.0), 

311 verdict=e.get("verdict", ""), 

312 confidence=e.get("confidence", ""), 

313 signal_scores=e.get("signal_scores", {}), 

314 btc_price=e.get("btc_price"), 

315 recorded_at=e.get("recorded_at", ""), 

316 ) 

317 ) 

318 except (KeyError, TypeError) as exc: 

319 logger.debug("항목 파싱 실패 (date=%s): %s", e.get("date"), exc) 

320 return sorted(snapshots, key=lambda s: s.date, reverse=True)

322 def format_accuracy_summary(self, lookback_days: int = 30) -> str: 

323 """정확도 요약을 마크다운 문자열로 반환한다. 

324 

325 히스토리가 없거나 평가 불가 시 빈 문자열을 반환한다. 

326 

327 Args: 

328 lookback_days: 분석할 최근 일수. 

329 

330 Returns: 

331 마크다운 형식 정확도 요약 문자열. 

332 """ 

333 report = self.accuracy_report(lookback_days=lookback_days) 

334 if report.total_predictions == 0: 

335 return "" 

336 

337 lines = [ 

338 f"### 📊 신호 예측 정확도 (최근 {lookback_days}일)", 

339 "", 

340 f"- **전체 승률**: {report.win_rate:.1%} ({report.correct_predictions}/{report.total_predictions})", 

341 ] 

342 

343 # 버딕트별 승률 

344 if report.by_verdict: 

345 lines.append("- **버딕트별 정확도**:") 

346 for verdict, stats in report.by_verdict.items(): 

347 lines.append(f" - {verdict}: {stats['win_rate']:.1%} ({stats['correct']}/{stats['total']})") 

348 

349 # 최근 5일 예측 이력 

350 recent5 = report.recent_accuracy[:5] 

351 if recent5: 

352 lines.append("") 

353 lines.append("| 날짜 | 예측 | 점수 | 실제 방향 | 결과 |") 

354 lines.append("|------|------|------|-----------|------|") 

355 for r in recent5: 

356 chg = f"{r.actual_price_change_pct:+.2f}%" if r.actual_price_change_pct is not None else "—" 

357 result_icon = "✅" if r.correct else ("❌" if r.correct is False else "—") 

358 lines.append( 

359 f"| {r.date} | {r.predicted_verdict} | {r.predicted_score:.1f} " 

360 f"| {r.actual_direction or '—'} ({chg}) | {result_icon} |" 

361 ) 

362 

363 return "\n".join(lines)

367 def _update_previous_accuracy(self, today_date: str, today_btc_price: float) -> None: 

368 """오늘 BTC 가격을 이용해 전날 예측 정확도를 업데이트한다. 

369 

370 전날 항목을 찾아 btc_price 대비 변동률을 계산하고, 

371 verdict 예측 방향과 비교하여 correct 필드를 기록한다. 

372 

373 Args: 

374 today_date: 오늘 날짜 YYYY-MM-DD. 

375 today_btc_price: 오늘 BTC 가격 (USD). 

376 """ 

377 try: 

378 today_dt = datetime.strptime(today_date, "%Y-%m-%d").replace(tzinfo=UTC) 

379 yesterday_str = (today_dt - timedelta(days=1)).strftime("%Y-%m-%d") 

380 except ValueError: 

381 return 

382 

383 prev_entry = next((e for e in self._entries if e.get("date") == yesterday_str), None) 

384 if prev_entry is None: 

385 logger.debug("전날 항목 없음 (date=%s)", yesterday_str) 

386 return 

387 

388 prev_btc = prev_entry.get("btc_price") 

389 if prev_btc is None or prev_btc <= 0: 

390 logger.debug("전날 BTC 가격 없음, 정확도 평가 불가") 

391 return 

392 

393 change_pct = ((today_btc_price - prev_btc) / prev_btc) * 100 

394 actual_dir = _price_direction(change_pct) 

395 

396 predicted_verdict = prev_entry.get("verdict", "") 

397 predicted_direction = _verdict_to_direction(predicted_verdict) 

398 

399 correct: Optional[bool] = None 

400 if predicted_direction is not None: 

401 correct = predicted_direction == actual_dir 

402 

403 prev_entry["accuracy"] = { 

404 "predicted_verdict": predicted_verdict, 

405 "predicted_score": prev_entry.get("composite_score", 0.0), 

406 "actual_price_change_pct": round(change_pct, 4), 

407 "actual_direction": actual_dir, 

408 "correct": correct, 

409 "evaluated_at": datetime.now(UTC).strftime("%Y-%m-%dT%H:%M:%SZ"), 

410 } 

411 

412 result_label = "✅ 정확" if correct else ("❌ 오류" if correct is False else "평가불가") 

413 logger.info( 

414 "정확도 업데이트 (date=%s): 예측=%s 실제=%s(%+.2f%%) → %s", 

415 yesterday_str, 

416 predicted_verdict, 

417 actual_dir, 

418 change_pct, 

419 result_label, 

420 )
