1"""Backfill missing accuracy fields in _state/signal_history.json. 

2 

3Entries recorded before Phase 3 signal_tracker migration may be missing the 

4`accuracy` nested dict. This script normalises those entries by inserting a 

5stub accuracy structure that is schema-compatible with SignalTracker. 

6 

7The stub uses the entry's own verdict/score and leaves price-dependent fields 

8(actual_price_change_pct, actual_direction, correct) as None, since historical 

9BTC prices are not re-fetched here. btc_price=null entries are left as-is 

10(nullable, already handled by _SIGNAL_SCHEMA.allow_null_fields). 

11 

12Usage: 

13 python scripts/backfill_signal_history_accuracy.py # dry-run (default) 

14 python scripts/backfill_signal_history_accuracy.py --apply # write changes 

15""" 

16 

17from __future__ import annotations 

18 

19import argparse 

20import os 

21import sys 

22from datetime import UTC, datetime 

23from pathlib import Path 

24from typing import Any 

25 

26# Allow running from repo root or scripts/ directory 

27_SCRIPTS_DIR = os.path.dirname(os.path.abspath(__file__)) 

28_REPO_ROOT = os.path.abspath(os.path.join(_SCRIPTS_DIR, "..")) 

29if _SCRIPTS_DIR not in sys.path: 

30 sys.path.insert(0, _SCRIPTS_DIR)

32from common.config import setup_logging # noqa: E402 

33from common.time_series_state import TimeSeriesSchema, TimeSeriesStore # noqa: E402 

34 

35logger = setup_logging("backfill_signal_history_accuracy") 

36 

37_HISTORY_FILE = os.path.join(_REPO_ROOT, "_state", "signal_history.json") 

38_NOW_UTC = datetime.now(UTC).strftime("%Y-%m-%dT%H:%M:%SZ") 

39 

40# Mirrors _SIGNAL_SCHEMA from signal_tracker.py — kept local to avoid import 

41# side-effects (signal_tracker imports signal_composer which may not be 

42# available in all environments). 

43_SIGNAL_SCHEMA = TimeSeriesSchema( 

44 required_fields=["date"], 

45 numeric_fields={}, 

46 date_field="date", 

47 date_format="%Y-%m-%d", 

48 max_entries=365, 

49 allow_null_fields=["btc_price"], 

50 extra_fields_allowed=True, 

51)

57def _make_accuracy_stub(entry: dict[str, Any]) -> dict[str, Any]: 

58 """Return a minimal accuracy dict compatible with signal_tracker schema. 

59 

60 Fields that require a subsequent day's BTC price are set to None. 

61 The stub is valid for SignalTracker.accuracy_report() — it skips records 

62 where correct is None. 

63 

64 Args: 

65 entry: A signal_history entry dict. 

66 

67 Returns: 

68 accuracy dict with all required keys. 

69 """ 

70 return { 

71 "predicted_verdict": entry.get("verdict", ""), 

72 "predicted_score": entry.get("composite_score", 0.0), 

73 "actual_price_change_pct": None, 

74 "actual_direction": None, 

75 "correct": None, 

76 "evaluated_at": _NOW_UTC, 

77 "backfilled_accuracy_stub": True, 

78 } 

79

84def find_missing_accuracy_indices(entries: list[dict[str, Any]]) -> list[int]: 

85 """Return list of indices where accuracy field is absent. 

86 

87 Args: 

88 entries: List of signal_history records. 

89 

90 Returns: 

91 List of integer indices (0-based) of entries missing accuracy. 

92 """ 

93 return [i for i, e in enumerate(entries) if "accuracy" not in e]

96def backfill(history_file: str = _HISTORY_FILE, dry_run: bool = True) -> int: 

97 """Main backfill routine. 

98 

99 Args: 

100 history_file: Path to signal_history.json. 

101 dry_run: If True, report changes without writing. Default True. 

102 

103 Returns: 

104 0 on success, 1 on error. 

105 """ 

106 path = Path(history_file) 

107 

108 if not path.exists(): 

109 logger.info("파일 없음, 종료: %s", history_file) 

110 print(f"File not found (nothing to do): {history_file}") 

111 return 0 

112 

113 store = TimeSeriesStore(path, _SIGNAL_SCHEMA, logger) 

114 entries: list[dict[str, Any]] = store.load(validate=False) 

115 

116 if not entries: 

117 logger.info("빈 히스토리, 종료") 

118 print("Empty history file. Nothing to backfill.") 

119 return 0 

120 

121 missing_indices = find_missing_accuracy_indices(entries) 

122 

123 if not missing_indices: 

124 logger.info("accuracy 누락 항목 없음. 완료.") 

125 print("No missing accuracy entries found. Nothing to backfill.") 

126 return 0 

127 

128 missing_dates = [entries[i].get("date", f"idx={i}") for i in missing_indices] 

129 print(f"누락된 accuracy: {len(missing_indices)}건") 

130 print(f"수정 대상 idx 리스트: {missing_indices}") 

131 print(f"수정 대상 날짜: {missing_dates}") 

132 print() 

133 

134 if dry_run: 

135 for i in missing_indices: 

136 entry = entries[i] 

137 date = entry.get("date", f"idx={i}") 

138 btc = entry.get("btc_price") 

139 btc_note = f"btc_price={btc}" if btc is not None else "btc_price=null (nullable, kept)" 

140 print(f" DRY-RUN WOULD PATCH idx={i} date={date} ({btc_note})") 

141 print() 

142 print("DRY-RUN: no changes written. Re-run with --apply to write.") 

143 return 0 

144 

145 # Apply: insert stubs for missing entries 

146 for i in missing_indices: 

147 entries[i]["accuracy"] = _make_accuracy_stub(entries[i]) 

148 logger.info( 

149 "accuracy 스텁 삽입: idx=%d date=%s", 

150 i, 

151 entries[i].get("date", "?"), 

152 ) 

153 

154 store._write_atomic(entries) 

155 print(f"APPLIED: {len(missing_indices)}건 accuracy 스텁 삽입, 파일 원자적 쓰기 완료.") 

156 

157 # Post-apply validation 

158 issues = store.validate(entries) 

159 errors = [iss for iss in issues if iss.severity == "error"] 

160 if errors: 

161 logger.warning("사후 검증 오류 %d건", len(errors)) 

162 for iss in errors: 

163 logger.warning(" [%s] idx=%d %s", iss.code, iss.index, iss.message) 

164 print(f"WARNING: {len(errors)} validation error(s) after apply (see logs).") 

165 else: 

166 print("VALIDATION: 0 errors after apply.") 

167 

168 # Confirm no remaining missing accuracy 

169 remaining = find_missing_accuracy_indices(entries) 

170 if remaining: 

171 print(f"WARNING: {len(remaining)} entries still missing accuracy: {remaining}") 

172 else: 

173 print("VALIDATION: 0 missing accuracy entries remain.") 

174 

175 return 0
