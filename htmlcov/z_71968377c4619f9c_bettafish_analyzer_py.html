104 """수치 점수(-1.0 ~ 1.0)를 버딕트 문자열로 변환한다."""

114def _confidence_from_agreement(agree_count: int, total: int) -> str: 

115 """일치 개수 / 전체 관점 수로 신뢰도를 결정한다.""" 

116 if total == 0: 

117 return "low" 

118 ratio = agree_count / total 

119 if ratio >= 0.75: 

120 return "high" 

121 if ratio >= 0.5: 

122 return "medium" 

123 return "low"

143 def analyze(self, composite_result: Any | None) -> dict[str, Any]: 

144 """CompositeResult를 분석해 데이터 관점 결과를 반환한다. 

145 

146 Args: 

147 composite_result: signal_composer.CompositeResult 인스턴스. 

148 None이면 기본값(중립)을 반환한다. 

149 

150 Returns: 

151 dict: narrative(str), verdict(str), score(float), 

152 strongest_bullish(str), strongest_bearish(str), 

153 agreement_ratio(float) 포함. 

154 """ 

155 if composite_result is None: 

156 logger.warning("DataPerspective: composite_result 없음 → 기본값 반환") 

157 return { 

158 "narrative": "데이터 신호가 제공되지 않아 정량 분석을 수행할 수 없습니다. 시장 중립 상태로 가정합니다.", 

159 "verdict": "중립", 

160 "score": 0.0, 

161 "strongest_bullish": "", 

162 "strongest_bearish": "", 

163 "agreement_ratio": 0.0, 

164 "risk_factors": [], 

165 "opportunities": [], 

166 } 

167 

168 score = getattr(composite_result, "score", 50.0) 

169 verdict = getattr(composite_result, "verdict", "중립") 

170 signal_results = getattr(composite_result, "signal_results", []) 

171 agreement_count = getattr(composite_result, "agreement_count", 0) 

172 total_signals = getattr(composite_result, "total_signals", 0) 

173 confidence_label = getattr(composite_result, "confidence_label", "낮음") 

174 

175 agreement_ratio = agreement_count / total_signals if total_signals > 0 else 0.0 

176 

177 # 가장 강한 강세/약세 신호 식별 

178 bullish_signals = sorted( 

179 [s for s in signal_results if getattr(s, "verdict", "") == "강세"], 

180 key=lambda s: getattr(s, "normalized", 0.5), 

181 reverse=True, 

182 ) 

183 bearish_signals = sorted( 

184 [s for s in signal_results if getattr(s, "verdict", "") == "약세"], 

185 key=lambda s: getattr(s, "normalized", 0.5), 

186 ) 

187 

188 strongest_bullish = bullish_signals[0].name if bullish_signals else "" 

189 strongest_bearish = bearish_signals[0].name if bearish_signals else "" 

190 

191 # 신호 요약 문자열 

192 signal_summary_parts = [] 

193 for sr in signal_results[:3]: 

194 name = getattr(sr, "name", "") 

195 raw = getattr(sr, "raw_display", "") 

196 sv = getattr(sr, "verdict", "") 

197 arrow = getattr(sr, "trend_arrow", "") 

198 signal_summary_parts.append(f"{name}({raw}{arrow}: {sv})") 

199 signal_summary = ", ".join(signal_summary_parts) if signal_summary_parts else "신호 없음" 

200 

201 # 서사 생성 

202 narrative = self._build_narrative( 

203 score=score, 

204 verdict=verdict, 

205 signal_summary=signal_summary, 

206 strongest_bullish=strongest_bullish, 

207 strongest_bearish=strongest_bearish, 

208 agreement_ratio=agreement_ratio, 

209 confidence_label=confidence_label, 

210 total_signals=total_signals, 

211 ) 

212 

213 # 리스크/기회 추출 

214 risk_factors = [s.name for s in bearish_signals[:3]] if bearish_signals else [] 

215 opportunities = [s.name for s in bullish_signals[:3]] if bullish_signals else [] 

216 

217 logger.info( 

218 "DataPerspective 분석 완료: verdict=%s, score=%.1f, agreement=%.0f%%", 

219 verdict, 

220 score, 

221 agreement_ratio * 100, 

222 ) 

223 

224 return { 

225 "narrative": narrative, 

226 "verdict": verdict, 

227 "score": score, 

228 "strongest_bullish": strongest_bullish, 

229 "strongest_bearish": strongest_bearish, 

230 "agreement_ratio": agreement_ratio, 

231 "risk_factors": risk_factors, 

232 "opportunities": opportunities, 

233 }

290class SentimentPerspective: 

291 """뉴스 심리와 토픽 데이터를 분석해 서사를 생성하는 관점 모듈. 

292 

293 MindSpider의 키워드/클러스터 결과를 입력받아: 

294 - 토픽별 감성 분포 분석 

295 - 심리-키워드 패턴 식별 (어떤 토픽이 긍정/부정 감성을 유발하는지) 

296 - 심리 관점 서사 문단 생성 

297 """ 

298 

299 def analyze( 

300 self, 

301 topic_clusters: list[Any] | None, 

302 keywords: list[dict] | None, 

303 ) -> dict[str, Any]: 

304 """토픽 클러스터와 키워드를 분석해 심리 관점 결과를 반환한다. 

305 

306 Args: 

307 topic_clusters: mindspider.TopicCluster 리스트. None이면 기본값 반환. 

308 keywords: extract_keywords() 반환 딕셔너리 리스트. 

309 각 항목은 'keyword', 'sentiment', 'count' 키를 포함. 

310 

311 Returns: 

312 dict: narrative(str), verdict(str), sentiment_score(float), 

313 positive_topics(list), negative_topics(list), 

314 dominant_bearish_kws(list), dominant_bullish_kws(list), 

315 risk_factors(list), opportunities(list) 포함. 

316 """ 

317 clusters = topic_clusters or [] 

318 kws = keywords or [] 

319 

320 if not clusters and not kws: 

321 logger.warning("SentimentPerspective: 입력 데이터 없음 → 기본값 반환") 

322 return { 

323 "narrative": "뉴스 심리 데이터가 제공되지 않아 심리 분석을 수행할 수 없습니다. 시장 중립 상태로 가정합니다.", 

324 "verdict": "중립", 

325 "sentiment_score": 0.0, 

326 "positive_topics": [], 

327 "negative_topics": [], 

328 "dominant_bullish_kws": [], 

329 "dominant_bearish_kws": [], 

330 "risk_factors": [], 

331 "opportunities": [], 

332 } 

333 

334 # 토픽별 감성 분류 

335 positive_topics: list[str] = [] 

336 negative_topics: list[str] = [] 

337 total_score = 0.0 

338 n_clusters = len(clusters) 

339 

340 for cluster in clusters: 

341 topic_name = getattr(cluster, "topic_name", "") 

342 sentiment = getattr(cluster, "sentiment_score", 0.0) 

343 total_score += sentiment 

344 if sentiment > 0.05: 

345 positive_topics.append(topic_name) 

346 elif sentiment < -0.05: 

347 negative_topics.append(topic_name) 

348 

349 avg_sentiment = total_score / n_clusters if n_clusters > 0 else 0.0 

350 

351 # 키워드별 강세/약세 분류 

352 bullish_kws = [kw["keyword"] for kw in kws if kw.get("sentiment") == "bullish"][:5] 

353 bearish_kws = [kw["keyword"] for kw in kws if kw.get("sentiment") == "bearish"][:5] 

354 

355 # 감성 점수 → 버딕트 

356 if avg_sentiment > 0.15: 

357 verdict = "강세" 

358 elif avg_sentiment < -0.15: 

359 verdict = "약세" 

360 elif abs(avg_sentiment) <= 0.05 and positive_topics and negative_topics: 

361 verdict = "혼조" 

362 else: 

363 verdict = "중립" 

364 

365 narrative = self._build_narrative( 

366 n_clusters=n_clusters, 

367 avg_sentiment=avg_sentiment, 

368 verdict=verdict, 

369 positive_topics=positive_topics, 

370 negative_topics=negative_topics, 

371 bullish_kws=bullish_kws, 

372 bearish_kws=bearish_kws, 

373 ) 

374 

375 # 리스크: 부정 토픽 + 약세 키워드 

376 risk_factors = list(negative_topics[:2]) + [f"키워드 '{kw}'" for kw in bearish_kws[:2]] 

377 opportunities = list(positive_topics[:2]) + [f"키워드 '{kw}'" for kw in bullish_kws[:2]] 

378 

379 logger.info( 

380 "SentimentPerspective 분석 완료: verdict=%s, avg_sentiment=%.3f, 클러스터=%d개", 

381 verdict, 

382 avg_sentiment, 

383 n_clusters, 

384 ) 

385 

386 return { 

387 "narrative": narrative, 

388 "verdict": verdict, 

389 "sentiment_score": round(avg_sentiment, 3), 

390 "positive_topics": positive_topics, 

391 "negative_topics": negative_topics, 

392 "dominant_bullish_kws": bullish_kws, 

393 "dominant_bearish_kws": bearish_kws, 

394 "risk_factors": risk_factors, 

395 "opportunities": opportunities, 

396 }

412 pos_n = len(positive_topics) 

413 neg_n = len(negative_topics) 

414 sentiment_sign = "+" if avg_sentiment >= 0 else "" 

415 if n_clusters > 0: 

416 lines.append(

419 ) 

420 else:

422

424 if bearish_kws and bullish_kws: 

425 bearish_str = _format_list_inline(bearish_kws[:3], " · ")

430 ) 

431 elif bearish_kws: 

432 bearish_str = _format_list_inline(bearish_kws[:3], " · ") 

433 lines.append( 

434 f"'{bearish_str}' 관련 뉴스가 전반적인 하방 심리를 주도하고 있으며, " 

435 f"반전을 이끌 긍정적 재료는 제한적입니다." 

436 ) 

437 elif bullish_kws: 

438 bullish_str = _format_list_inline(bullish_kws[:3], " · ") 

439 lines.append( 

440 f"'{bullish_str}' 관련 뉴스가 시장 낙관론을 뒷받침하고 있으며, 부정적 재료는 현재 두드러지지 않습니다." 

441 ) 

442 

443 # 셋째 문장: 심리 수렴/발산 코멘트

478 def analyze(self, macro_data: dict[str, Any] | None) -> dict[str, Any]:

595 verdict, 

596 macro_score_norm, 

597 len(tailwinds), 

598 len(headwinds), 

599 )

705 def decompose(self, verdict: str, signals: dict[str, Any]) -> list[str]: 

706 """시장 상황을 하위 질문으로 분해한다. 

707 

708 Args:

710 signals: 신호 데이터 딕셔너리 (composite_result, macro_data 등). 

711 

712 Returns: 

713 list[str]: 분석할 하위 질문 목록. 

714 """

716 

717 # 신호 데이터에 따라 추가 질문 삽입 

718 extra = self._generate_market_sub_questions(signals) 

719 combined = base_questions + [q for q in extra if q not in base_questions] 

720 return combined[:5] # 최대 5개 하위 질문
