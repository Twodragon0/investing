84def _verdict_to_score(verdict: str) -> float: 

85 """버딕트를 -1.0 ~ 1.0 수치 점수로 변환한다.""" 

86 mapping = {"강세": 1.0, "약세": -1.0, "중립": 0.0, "혼조": 0.0} 

87 return mapping.get(verdict, 0.0)

90def _score_to_verdict(score: float) -> str: 

91 """수치 점수(-1.0 ~ 1.0)를 버딕트 문자열로 변환한다.""" 

92 if score >= 0.4: 

93 return "강세" 

94 if score <= -0.4: 

95 return "약세" 

96 if abs(score) < 0.15: 

97 return "중립" 

98 return "혼조"

101def _confidence_from_agreement(agree_count: int, total: int) -> str: 

102 """일치 개수 / 전체 관점 수로 신뢰도를 결정한다.""" 

103 if total == 0: 

104 return "low" 

105 ratio = agree_count / total 

106 if ratio >= 0.75: 

107 return "high" 

108 if ratio >= 0.5: 

109 return "medium" 

110 return "low"
