19# Trailing junk fragments that some sources/translations append to article 

20# descriptions (ad slugs, mangled "관련 X" tails). Anchored at the END only so 

21# legitimate body content is never touched — unlike the whole-description 

22# boilerplate reject in enrichment._is_site_boilerplate. 

23# 

24# Korean noun set is deliberately limited to terms that never carry article 

25# meaning in a news summary (광고/정보/뉴스/주소/홍보). "관련 보도" and similar 

26# are EXCLUDED because enrichment._analyze_*_title emits them as a legitimate 

27# synthetic suffix. An optional 2–5 char lead-in absorbs mangled fragments 

28# like "급락 " / "등급락 " so "...경고했습니다. 급락 관련 주소." → "...경고했습니다.". 

29# English markers are restricted to multi-word slugs to avoid truncating a real 

30# sentence that merely ends in "sponsored"/"advertisement". 

31_TRAILING_ARTIFACT_RE = re.compile( 

32 r"\s*(?:" 

33 r"(?:[가-힣]{2,5}\s+)?관련\s?(?:광고|정보|뉴스|주소|홍보)" 

34 r"|read\s+more|sponsored\s+content" 

35 r")\s*[.。!?]?\s*$", 

36 re.IGNORECASE, 

37)

40def _strip_trailing_artifacts(text: str) -> str: 

41 """Strip trailing ad/boilerplate fragments left at the tail of a description. 

42 

43 Removes only end-anchored junk (e.g. "... 가격이 폭락했습니다. 등급락 관련정보.") 

44 and loops so stacked tails are fully cleared. Returns the cleaned text with 

45 surrounding whitespace stripped. 

46 """ 

47 if not text: 

48 return text 

49 prev = None 

50 while prev != text: 

51 prev = text 

52 text = _TRAILING_ARTIFACT_RE.sub("", text).rstrip() 

53 return text.strip()

56def _truncate_sentence(text: str, max_len: int = 300) -> str: 

57 """Truncate text at the nearest sentence boundary within max_len. 

58 

59 Uses forward-search strategy to find the first complete sentence, 

60 unlike utils.truncate_sentence which uses backward-search. 

61 Returns empty string if text is too short to be useful. 

62 """ 

63 text = text.strip() 

64 if not text or len(text) < 15: 

65 return "" 

66 if len(text) <= max_len: 

67 return text 

68 

69 _SENTENCE_ENDS = [ 

70 "다. ", 

71 "요. ", 

72 "음. ", 

73 "됩니다. ", 

74 "입니다. ", 

75 "습니다. ", 

76 "했다. ", 

77 "됐다. ", 

78 "였다. ", 

79 "합니다. ", 

80 "했습니다. ", 

81 "겠습니다. ", 

82 "봅니다. ", 

83 "。", 

84 ". ", 

85 "! ", 

86 "? ", 

87 ] 

88 best_idx = -1 

89 for sep in _SENTENCE_ENDS: 

90 idx = text.find(sep, 20) 

91 if 20 < idx < max_len: 

92 candidate = idx + len(sep) 

93 if candidate > best_idx: 

94 best_idx = candidate 

95 

96 if best_idx > 20: 

97 return text[:best_idx].strip() 

98 return _truncate_sentence_util(text, max_length=max_len)

101def _favicon_url(link: str) -> str: 

102 """Return a Google Favicon API URL for the domain of *link*. 

103 

104 Falls back to empty string if the link cannot be parsed. Uses sz=128 

105 for sharper rendering on retina displays. 

106 """ 

107 if not link: 

108 return "" 

109 try: 

110 from urllib.parse import urlparse 

111 

112 parsed = urlparse(link) 

113 domain = parsed.netloc or parsed.hostname or "" 

114 if domain: 

115 return f"https://www.google.com/s2/favicons?domain={domain}&sz=128" 

116 except Exception: # noqa: S110 

117 return "" 

118 return ""

121def _best_favicon_link(item: dict) -> str: 

122 """Pick the best link for favicon extraction. 

123 

124 Prefers the resolved article URL (``original_url``) over the raw link, 

125 so Google News redirect URLs use the actual publisher's favicon 

126 (e.g., cnbc.com) instead of news.google.com's generic favicon. 

127 """ 

128 original = item.get("original_url", "") 

129 if original and "news.google.com" not in original and "google." not in original.split("/")[2:3][0:1]: 

130 return original 

131 link = item.get("link", "") 

132 if link and "news.google.com" not in link: 

133 return link 

134 return original or link
