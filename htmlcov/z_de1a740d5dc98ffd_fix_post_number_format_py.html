1#!/usr/bin/env python3 

2"""유럽식 숫자 포맷 깨짐을 미국식으로 일괄 보정. 

3 

4_posts/ 디렉토리에서 최근 N일 내 .md 파일을 스캔하여 

5유럽식 숫자 포맷(예: "BTC$71.018,21")을 미국식(예: "BTC $71,018.21")으로 치환. 

6 

7사용법: 

8 python scripts/fix_post_number_format.py --days 30 # dry-run (기본) 

9 python scripts/fix_post_number_format.py --days 30 --apply # 실제 적용 

10""" 

11 

12import argparse 

13import re 

14import sys 

15from datetime import UTC, datetime, timedelta 

16from pathlib import Path 

17 

18# translator.py:563-565 와 동일한 정규식 

19_PATTERNS: list[tuple[re.Pattern, str]] = [ 

20 # "BTC$71.018,21" → "BTC $71,018.21" 

21 (re.compile(r"([A-Z]{2,5})\$(\d{1,3})\.(\d{3}),(\d{2})\b"), r"\1 $\2,\3.\4"), 

22 # "$71.018,21" → "$71,018.21" 

23 (re.compile(r"\$(\d{1,3})\.(\d{3}),(\d{2})\b"), r"$\1,\2.\3"), 

24] 

25 

26 

27def _fix_content(content: str) -> tuple[str, int]: 

28 """콘텐츠에 패턴 치환 적용. (수정된 내용, 치환 횟수) 반환.""" 

29 total_subs = 0 

30 for pattern, replacement in _PATTERNS: 

31 content, n = pattern.subn(replacement, content) 

32 total_subs += n 

33 return content, total_subs 

34 

35 

36def _post_mtime(path: Path) -> datetime: 

37 return datetime.fromtimestamp(path.stat().st_mtime, tz=UTC) 

38 

39 

40def scan_posts(posts_dir: Path, days: int) -> list[Path]: 

41 """mtime 기준 최근 N일 내 .md 파일 목록 반환.""" 

42 cutoff = datetime.now(tz=UTC) - timedelta(days=days) 

43 return sorted(p for p in posts_dir.glob("*.md") if _post_mtime(p) >= cutoff) 

44 

45 

46def main() -> int: 

47 parser = argparse.ArgumentParser(description="유럽식 숫자 포맷을 미국식으로 일괄 보정.") 

48 parser.add_argument("--days", type=int, default=30, help="스캔할 최근 일수 (기본: 30)") 

49 parser.add_argument("--apply", action="store_true", help="실제 파일 변경 (없으면 dry-run)") 

50 parser.add_argument( 

51 "--posts-dir", 

52 type=Path, 

53 default=Path("_posts"), 

54 help="_posts 디렉토리 경로 (기본: _posts)", 

55 ) 

56 args = parser.parse_args() 

57 

58 posts_dir = args.posts_dir 

59 if not posts_dir.exists(): 

60 print(f"오류: _posts 디렉토리를 찾을 수 없습니다: {posts_dir}", file=sys.stderr) 

61 return 2 

62 

63 mode = "적용" if args.apply else "DRY-RUN" 

64 print(f"[숫자 포맷 보정 | {mode}] 최근 {args.days}일 포스트 스캔 중...") 

65 

66 posts = scan_posts(posts_dir, args.days) 

67 print(f" 스캔 대상: {len(posts)}개 파일") 

68 

69 changed_files: list[tuple[Path, int]] = [] 

70 total_replacements = 0 

71 

72 for post in posts: 

73 content = post.read_text(encoding="utf-8", errors="ignore") 

74 new_content, subs = _fix_content(content) 

75 if subs == 0: 

76 continue 

77 changed_files.append((post, subs)) 

78 total_replacements += subs 

79 if args.apply: 

80 post.write_text(new_content, encoding="utf-8") 

81 print(f" [수정됨] {post.name} ({subs}건 치환)") 

82 else: 

83 print(f" [예정] {post.name} ({subs}건 치환 예정)") 

84 

85 print() 

86 if args.apply: 

87 print(f"완료: {len(changed_files)}개 파일 수정, 총 {total_replacements}건 치환.") 

88 else: 

89 print(f"DRY-RUN 결과: {len(changed_files)}개 파일 영향, 총 {total_replacements}건 치환 예정.") 

90 if changed_files: 

91 print("실제 적용하려면:") 

92 print(f" python scripts/fix_post_number_format.py --days {args.days} --apply") 

93 

94 return 0 

95 

96 

97if __name__ == "__main__": 

98 raise SystemExit(main())
