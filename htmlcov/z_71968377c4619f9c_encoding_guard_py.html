1"""Shared encoding guards for text fetched from external sources. 

2 

3Centralises the mojibake detector and recovery helper used by 

4``rss_fetcher.py`` and ``enrichment.py`` so that any description that 

5enters the post pipeline has a single, consistent sanitisation step. 

6 

7Mojibake typically arises when a remote server sends UTF-8 bytes but 

8declares ``Content-Type: text/html; charset=iso-8859-1`` (or similar), 

9causing ``requests`` to decode ``resp.text`` as Latin-1. Downstream the 

10text looks like ``ì£¼ëì ê¸°ì ììì`` — legitimate Korean never contains 

11such runs. 

12""" 

13 

14from __future__ import annotations 

15 

16import logging 

17import re 

18from typing import Any 

19 

20logger = logging.getLogger(__name__) 

21 

22# Runs of Latin-1 supplement + C1 control chars (U+0080–U+00FF) that 

23# indicate UTF-8 bytes were misinterpreted as Latin-1/CP1252 upstream. 

24# Legitimate Korean text never contains such runs; accented European text 

25# rarely exceeds 2 consecutive chars in this block. 

26_MOJIBAKE_RE = re.compile(r"[\u0080-\u00ff]{3,}") 

27 

28 

29def is_mojibake(text: str) -> bool: 

30 """Return True if *text* shows a mojibake pattern.""" 

31 if not text: 

32 return False 

33 return bool(_MOJIBAKE_RE.search(text)) 

34 

35 

36def sanitize_mojibake(text: str) -> str: 

37 """Return cleaned text, attempting Latin-1 → UTF-8 recovery when possible. 

38 

39 Strategy: 

40 1. Round-trip recover: re-encode as Latin-1 and decode as UTF-8. If 

41 the recovered text contains Hangul or CJK characters, return it. 

42 2. If recovery fails AND the text still has a long mojibake run, drop 

43 the description so downstream synthetic fallback can take over. 

44 """ 

45 if not text: 

46 return text 

47 # Fast path: no high-byte chars → nothing to recover. 

48 if not any(0x80 <= ord(c) <= 0xFF for c in text): 

49 return text 

50 # Attempt round-trip recovery: UTF-8 bytes misinterpreted as Latin-1. 

51 try: 

52 recovered = text.encode("latin-1", errors="strict").decode("utf-8", errors="strict") 

53 except (UnicodeEncodeError, UnicodeDecodeError): 

54 recovered = None 

55 if recovered and any(0xAC00 <= ord(c) <= 0xD7A3 or 0x4E00 <= ord(c) <= 0x9FFF for c in recovered): 

56 return recovered 

57 # Unrecoverable — if mojibake pattern is still present, drop the text. 

58 if _MOJIBAKE_RE.search(text): 

59 logger.warning("Description dropped due to unrecoverable mojibake") 

60 return "" 

61 return text 

62 

63 

64def force_utf8_if_mislabelled(response: Any) -> None: 

65 """Force ``response.encoding = 'utf-8'`` when the server labels the 

66 body as Latin-1 / ISO-8859-1 / ASCII. 

67 

68 Many Korean news endpoints (Google News RSS, certain CDNs) serve 

69 UTF-8 with an incorrect ``Content-Type`` header; without this nudge 

70 ``response.text`` comes back as mojibake and downstream extraction is 

71 corrupted at the source. 

72 """ 

73 encoding = getattr(response, "encoding", None) 

74 if encoding and encoding.lower() in ("iso-8859-1", "latin-1", "ascii"): 

75 response.encoding = "utf-8"
