323def _resolve_source_weight(source: str) -> float: 

324 """Return base weight for a source string using 6-type classification.""" 

325 # Lazy import to avoid circular dependency with markdown_utils 

326 try: 

327 from .markdown_utils import _classify_source # type: ignore[import] 

328 

329 src_type = _classify_source(source) 

330 weight = _SOURCE_TYPE_WEIGHTS.get(src_type, 1.0) 

331 except Exception: 

332 weight = 1.0 

333 

334 # Fallback: per-domain override (catches named sources not yet in _SOURCE_RULES) 

335 src_low = source.lower() 

336 for domain, w in _DOMAIN_WEIGHTS.items(): 

337 if domain in src_low: 

338 weight = max(weight, w) 

339 break 

340 

341 # Clip to (0.5, 2.5) 

342 lo, hi = WEIGHTS["source_base"] 

343 return max(lo, min(hi, weight))

346def _has_institution(text: str) -> bool: 

347 """Return True when text mentions a regulatory / institutional entity.""" 

348 return any(_word_present(term, text) for term in _ORG_ENTITY_TERMS)

351def _word_present(term: str, text: str) -> bool: 

352 """Check presence with light word-boundary handling for mixed ko/en.""" 

353 if not term: 

354 return False 

355 # English: use \b word boundaries 

356 if term.isascii(): 

357 pattern = re.compile(r"\b" + re.escape(term) + r"\b", re.IGNORECASE) 

358 return bool(pattern.search(text)) 

359 # Korean: check surrounding chars are whitespace / punctuation / start-end 

360 idx = text.find(term) 

361 if idx == -1: 

362 return False 

363 pre_ok = idx == 0 or not text[idx - 1].isalpha() 

364 post_idx = idx + len(term) 

365 post_ok = post_idx >= len(text) or not text[post_idx].isalpha() 

366 return pre_ok and post_ok

369def _detect_sentiment(text: str) -> Literal["pos", "neg", "neu"]: 

370 """Classify sentiment as pos / neg / neu from keyword presence.""" 

371 pos_hits = sum(1 for kw in _SENTIMENT_POS if kw in text.lower()) 

372 neg_hits = sum(1 for kw in _SENTIMENT_NEG if kw in text.lower()) 

373 if pos_hits > neg_hits: 

374 return "pos" 

375 if neg_hits > pos_hits: 

376 return "neg" 

377 return "neu"

380def _downgrade_one(level: str) -> str: 

381 """Demote risk level by one tier.""" 

382 return {"critical": "elevated", "elevated": "moderate", "moderate": "low", "low": "low"}[level]

385def _item_text(item: dict[str, Any]) -> str: 

386 """Build combined text for signal extraction (title-first, no duplication).""" 

387 title = item.get("title", "") 

388 title_original = item.get("title_original", "") 

389 description = item.get("description", "") 

390 # Avoid counting title twice when title_original is the same (F5 mitigation) 

391 if title_original.strip().lower() == title.strip().lower(): 

392 title_original = "" 

393 return f"{title} {title_original} {description}"

401def extract_signals( 

402 item: dict[str, Any], 

403 sentiment_fn: Any = None, 

404 source_classifier: Any = None, 

405) -> RiskSignals: 

406 """Extract structured risk signals from a single news item dict. 

407 

408 sentiment_fn and source_classifier are optional callables for extension; 

409 built-in heuristics are used when they are None. 

410 """ 

411 source = item.get("source", "") 

412 text = _item_text(item) 

413 text_low = text.lower() 

414 

415 source_weight = _resolve_source_weight(source) if source_classifier is None else _resolve_source_weight(source) 

416 

417 has_amount = bool(_AMOUNT_RE.search(text)) 

418 has_inst = _has_institution(text) 

419 market_mech = any(kw in text_low for kw in _MARKET_MECHANISM) 

420 sec_exploit = any(kw in text_low for kw in _SECURITY_EXPLOIT_TERMS) 

421 is_entertain = any(_word_present(kw, text) for kw in _ENTERTAINMENT_KEYWORDS) 

422 

423 # Opinion detection 

424 is_opinion = any(_word_present(marker, text) for marker in _OPINION_MARKERS) 

425 # R2 mitigation: institution mention overrides opinion flag 

426 if has_inst: 

427 is_opinion = False 

428 

429 # Sentiment 

430 if sentiment_fn is not None: 

431 try: 

432 raw = sentiment_fn(item) 

433 sentiment: Literal["pos", "neg", "neu"] = raw if raw in ("pos", "neg", "neu") else "neu" 

434 except Exception: 

435 sentiment = _detect_sentiment(text) 

436 else: 

437 sentiment = _detect_sentiment(text) 

438 

439 return RiskSignals( 

440 source_weight=source_weight, 

441 has_amount=has_amount, 

442 has_institution=has_inst, 

443 market_mechanism=market_mech, 

444 is_opinion=is_opinion, 

445 is_entertainment=is_entertain, 

446 sentiment=sentiment, 

447 security_exploit=sec_exploit, 

448 )

451def score_item( 

452 item: dict[str, Any], 

453 item_id: str | None = None, 

454 sentiment_fn: Any = None, 

455 source_classifier: Any = None, 

456) -> ItemScore: 

457 """Compute a 0–10 impact score for a single news item.""" 

458 if item_id is None: 

459 item_id = item.get("id", item.get("url", item.get("title", "unknown"))) 

460 

461 signals = extract_signals(item, sentiment_fn=sentiment_fn, source_classifier=source_classifier) 

462 contributions: dict[str, float] = {} 

463 

464 # S1 — source base 

465 contributions["source"] = signals.source_weight 

466 

467 # S2 — concrete amount 

468 amt = WEIGHTS["amount"] if signals.has_amount else 0.0 

469 contributions["amount"] = amt 

470 

471 # S3 — institutional entity 

472 inst = WEIGHTS["institution"] if signals.has_institution else 0.0 

473 contributions["institution"] = inst 

474 

475 # S4 — market mechanism 

476 mech = WEIGHTS["market_mechanism"] if signals.market_mechanism else 0.0 

477 contributions["market_mechanism"] = mech 

478 

479 # S5 — opinion penalty 

480 op = WEIGHTS["opinion_penalty"] if signals.is_opinion else 0.0 

481 contributions["opinion_penalty"] = op 

482 

483 # S6 — entertainment penalty 

484 ent = WEIGHTS["entertainment_penalty"] if signals.is_entertainment else 0.0 

485 contributions["entertainment_penalty"] = ent 

486 

487 # S7 — sentiment 

488 if signals.sentiment == "pos": 

489 sent = WEIGHTS["sentiment_pos_penalty"] 

490 elif signals.sentiment == "neg": 

491 sent = WEIGHTS["sentiment_neg_bonus"] 

492 else: 

493 sent = 0.0 

494 contributions["sentiment"] = sent 

495 

496 raw_score = sum(contributions.values()) 

497 clipped = max(0.0, min(10.0, raw_score)) 

498 

499 return ItemScore( 

500 item_id=str(item_id), 

501 score=clipped, 

502 signals=signals, 

503 contributions=contributions, 

504 rule_overrides=[], 

505 )

508def apply_overrides( 

509 scores: list[ItemScore], 

510 base_level: str, 

511) -> tuple[str, list[str]]: 

512 """Apply rule-based overrides on top of weighted-sum base level. 

513 

514 Rule 6: market_mechanism + source_weight >= 1.5 → hard CRITICAL. 

515 Rule 7: majority of P0-like items are opinion → downgrade one tier. 

516 """ 

517 trace: list[str] = [] 

518 

519 # Rule 6 — hard critical for structural market events from credible sources 

520 hard_critical = [s for s in scores if s.signals.market_mechanism and s.signals.source_weight >= 1.5] 

521 if hard_critical: 

522 trace.append("rule_6_market_mechanism_hard_override") 

523 return "critical", trace 

524 

525 # Rule 6b — security exploit + amount + credible source → hard CRITICAL 

526 sec_critical = [ 

527 s for s in scores if s.signals.security_exploit and s.signals.has_amount and s.signals.source_weight >= 1.5 

528 ] 

529 if sec_critical: 

530 trace.append("rule_6b_security_exploit_hard_override") 

531 return "critical", trace 

532 

533 # Rule 7 — opinion majority among high-scoring items triggers downgrade 

534 p0_like = [s for s in scores if s.score >= P0_ITEM_THRESHOLD] 

535 if p0_like: 

536 opinion_ratio = sum(1 for s in p0_like if s.signals.is_opinion) / len(p0_like) 

537 if opinion_ratio > 0.5: 

538 trace.append(f"rule_7_opinion_ratio={opinion_ratio:.2f}_downgrade") 

539 return _downgrade_one(base_level), trace 

540 

541 return base_level, trace

544def classify_risk( 

545 items: list[dict[str, Any]], 

546 priority_items: dict[str, Any] | None = None, 

547 sentiment_fn: Any = None, 

548 source_classifier: Any = None, 

549) -> RiskVerdict: 

550 """Classify overall risk level for a list of news items. 

551 

552 Returns RiskVerdict with level, top_items (score-descending), and rule_trace. 

553 """ 

554 if not items: 

555 return RiskVerdict( 

556 level="low", 

557 reason="no items", 

558 top_items=[], 

559 aggregate_mean=0.0, 

560 rule_trace=[], 

561 ) 

562 

563 # Score all items 

564 scored: list[ItemScore] = [] 

565 for i, item in enumerate(items): 

566 item_id = item.get("id", item.get("url", f"item_{i}")) 

567 scored.append(score_item(item, item_id=item_id, sentiment_fn=sentiment_fn, source_classifier=source_classifier)) 

568 

569 # Sort descending by score 

570 sorted_scores = sorted(scored, key=lambda x: x.score, reverse=True) 

571 

572 # Aggregate mean of top-3 

573 top3 = sorted_scores[:3] 

574 aggregate_mean = sum(s.score for s in top3) / len(top3) if top3 else 0.0 

575 

576 # Determine base level from weighted-sum scores 

577 p0_items = [s for s in sorted_scores if s.score >= P0_ITEM_THRESHOLD] 

578 elevated_items = [s for s in sorted_scores if s.score >= ELEVATED_ITEM_THRESHOLD] 

579 

580 if len(p0_items) >= 3 and aggregate_mean >= CRITICAL_MEAN_TOP_3: 

581 base_level = "critical" 

582 reason = f"{len(p0_items)} high-score items, mean_top3={aggregate_mean:.2f}" 

583 elif p0_items: 

584 base_level = "elevated" 

585 reason = f"{len(p0_items)} items above P0 threshold" 

586 elif len(elevated_items) >= 5: 

587 base_level = "elevated" 

588 reason = f"{len(elevated_items)} items above elevated threshold" 

589 elif len(elevated_items) >= 2: 

590 base_level = "moderate" 

591 reason = f"{len(elevated_items)} items above elevated threshold" 

592 else: 

593 base_level = "low" 

594 reason = "no significant items" 

595 

596 # Apply rule overrides 

597 final_level, rule_trace = apply_overrides(sorted_scores, base_level) 

598 

599 logger.info( 

600 "risk_level=%s base=%s mean_top3=%.2f rules=%s", 

601 final_level, 

602 base_level, 

603 aggregate_mean, 

604 rule_trace, 

605 ) 

606 

607 return RiskVerdict( 

608 level=final_level, 

609 reason=reason, 

610 top_items=sorted_scores, 

611 aggregate_mean=aggregate_mean, 

612 rule_trace=rule_trace, 

613 )
