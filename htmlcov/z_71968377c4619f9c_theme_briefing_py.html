152 def generate_theme_subtitle(self, theme_key: str, articles: List[Dict[str, Any]]) -> str: 

153 """Generate a subtitle from the best article description for theme headings. 

154 

155 Unlike generate_single_theme_briefing which uses keyword analysis, 

156 this returns a direct description snippet from the top article, 

157 giving readers a concrete preview of the most important story. 

158 """ 

159 from .enrichment import _is_desc_duplicate_of_title 

160 from .summarizer import _is_generic_desc 

161 

162 if not articles: 

163 return "" 

164 

165 for article in articles[:5]: 

166 desc = _fix_mistranslations((article.get("description_ko") or article.get("description", "")).strip()) 

167 title = _fix_mistranslations(article.get("title_ko") or article.get("title", "")) 

168 # Skip descriptions that merely echo the title (incl. "title + source" 

169 # variants) — they produce a redundant italic subtitle above the card. 

170 if not desc or len(desc) < 20 or _is_desc_duplicate_of_title(desc, title): 

171 continue 

172 if _is_generic_desc(desc): 

173 continue 

174 sentences = re.split(r"(?<=[.!?。다요음])\s+", desc) 

175 snippet = sentences[0] if sentences else desc 

176 if len(snippet) > 120: 

177 snippet = snippet[:117].rsplit(" ", 1)[0] + "..." 

178 if len(snippet) >= 15: 

179 return snippet 

180 return ""

182 def generate_theme_briefing(self) -> str: 

183 """Generate combined theme briefings for all top themes. 

184 

185 Returns a section with 1-2 sentence briefings per theme, 

186 based on article descriptions. 

187 """ 

188 if len(self._ts.items) < 5: 

189 return "" 

190 

191 top_themes = self._ts.get_top_themes() 

192 if not top_themes: 

193 return "" 

194 

195 lines = ["## 테마별 브리핑\n"] 

196 has_content = False 

197 

198 for name, key, emoji, _count in top_themes: 

199 articles = self._ts.get_articles_for_theme(key) 

200 # Route through summarizer wrapper so tests that monkey-patch 

201 # ThemeSummarizer._generate_single_theme_briefing keep working. 

202 briefing = self._ts._generate_single_theme_briefing(key, articles) 

203 if briefing and briefing.strip() != name and len(briefing.strip()) > len(name): 

204 lines.append(f"- {emoji} **{name}**: {briefing}") 

205 has_content = True 

206 

207 if not has_content: 

208 return "" 

209 

210 lines.append("") 

211 return "\n".join(lines)
