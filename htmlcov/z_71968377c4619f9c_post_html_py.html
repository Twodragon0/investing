1"""Shared HTML builders for daily report posts. 

2 

3Centralises the three repeating blocks (stat-grid / alert-box / footer-meta) 

4that 7 daily-report collectors emit by hand. Goal: single source of truth so 

5the Designer-audited conventions (div-based stat-value, 3-coloured alert 

6classes, wm-footer-meta layout) stay consistent and future tweaks land in 

7one place instead of fanning out across 7 files. 

8""" 

9 

10from __future__ import annotations 

11 

12from typing import Iterable, Literal, Sequence 

13 

14AlertVariant = Literal["info", "warning", "urgent"] 

15 

16# Inline SVG icons (MIT-licensed shapes inspired by Heroicons/Material). Inline 

17# rather than asset files so each post stays self-contained, and the path 

18# inherits the surrounding `<strong>` color via fill="currentColor" — no 

19# extra CSS to keep dark/light theme parity. aria-hidden because the 

20# accompanying title text already conveys the semantic role (WCAG 1.1.1). 

21_SVG_TEMPLATE = ( 

22 '<svg xmlns="http://www.w3.org/2000/svg" width="16" height="16" ' 

23 'viewBox="0 0 24 24" fill="currentColor" aria-hidden="true" ' 

24 'focusable="false" class="alert-icon">{path}</svg>' 

25) 

26 

27_ALERT_ICON: dict[str, str] = { 

28 "info": _SVG_TEMPLATE.format( 

29 path='<path d="M12 2C6.48 2 2 6.48 2 12s4.48 10 10 10 10-4.48 10-10S17.52 2 12 2zm1 15h-2v-6h2v6zm0-8h-2V7h2v2z"/>' 

30 ), 

31 "warning": _SVG_TEMPLATE.format(path='<path d="M1 21h22L12 2 1 21zm12-3h-2v-2h2v2zm0-4h-2v-4h2v4z"/>'), 

32 "urgent": _SVG_TEMPLATE.format( 

33 path='<path d="M12 2C6.48 2 2 6.48 2 12s4.48 10 10 10 10-4.48 10-10S17.52 2 12 2zm1 15h-2v-2h2v2zm0-4h-2V7h2v6z"/>'

35}

38def stat_grid(items: Sequence[tuple[str, str]]) -> str: 

39 """Render a `<div class="stat-grid">` with `<div class="stat-item">` children. 

40 

41 `items` is a sequence of ``(value, label)`` tuples. Empty input returns ``""`` 

42 so callers can chain without conditional wrappers. 

43 """ 

44 if not items: 

45 return "" 

46 cells = "".join( 

47 f'<div class="stat-item"><div class="stat-value">{value}</div><div class="stat-label">{label}</div></div>' 

48 for value, label in items 

49 ) 

50 return f'<div class="stat-grid">{cells}</div>'

53def alert_box(title: str, bullets: Iterable[str], variant: AlertVariant = "info") -> str: 

54 """Render a `<div class="alert-box alert-{variant}">` callout. 

55 

56 `bullets` items are inserted as raw `<li>` content (caller controls inline 

57 HTML/markdown). Empty bullets list returns ``""``. A variant-specific icon 

58 is prepended to the title so the callout's role stays legible without 

59 relying on the border color (WCAG 1.4.1). 

60 """ 

61 bullet_list = list(bullets) 

62 if not bullet_list: 

63 return "" 

64 items_html = "".join(f"<li>{b}</li>" for b in bullet_list) 

65 icon = _ALERT_ICON[variant] 

66 return f'<div class="alert-box alert-{variant}"><strong>{icon} {title}</strong><ul>{items_html}</ul></div>'

69def summary_intro( 

70 date: str, 

71 label: str, 

72 headline: str | None, 

73 *, 

74 tag: str | None = None, 

75 detail: str = "", 

76) -> str: 

77 """Render the lead paragraph that becomes ``page.excerpt``. 

78 

79 Standard format used by daily-report collectors: 

80 ``**{date}** {label}: **{headline}** ({tag}). {detail}\\n`` 

81 

82 ``tag`` is the short qualifier shown in parentheses after the headline. 

83 It is purposefully generic — callers use it for source attribution 

84 ("GDELT", "Google News"), theme classification ("군사/분쟁"), or any 

85 other one-token context cue. Pass ``None`` to omit the parens entirely. 

86 

87 Headline-missing fallback: 

88 ``**{date}** {label} — {detail}\\n`` 

89 

90 The trailing newline is included so callers can prepend the result directly 

91 to a section body without manual spacing. 

92 """ 

93 if headline: 

94 tag_part = f" ({tag})" if tag else "" 

95 return f"**{date}** {label}: **{headline}**{tag_part}. {detail}\n" 

96 if detail: 

97 return f"**{date}** {label} — {detail}\n" 

98 return f"**{date}** {label}.\n"

101def footer_meta(timestamp: str, sources: str | Sequence[str]) -> str: 

102 """Render the `<div class="wm-footer-meta">` post footer. 

103 

104 `timestamp` is rendered as-is (caller decides locale/format). `sources` 

105 accepts either a pre-joined string or an iterable of source names. 

106 """ 

107 if not isinstance(sources, str): 

108 sources = ", ".join(s for s in sources if s) 

109 return f'<div class="wm-footer-meta"><span>수집 시각: {timestamp}</span><span>소스: {sources or "N/A"}</span></div>'
