1#!/usr/bin/env python3 

2"""Clean translation cache by re-applying postprocess fixes. 

3 

4Runs _postprocess_translation() on all cached values to fix artifacts 

5introduced by past translator bugs. Safe to run repeatedly — idempotent. 

6 

7Usage: 

8 python scripts/clean_translation_cache.py # apply fixes 

9 python scripts/clean_translation_cache.py --dry-run # preview only 

10""" 

11 

12import argparse 

13import json 

14import os 

15import re 

16import sys 

17 

18sys.path.insert(0, os.path.join(os.path.dirname(__file__))) 

19 

20from common.translator import _CACHE_PATH, _postprocess_translation 

21 

22# Cached values whose synthetic "...관련 보도." suffix was mutated by Google 

23# Translate into 통보/알림/공지/안내/보상/경고 (see enrichment._split_synthetic_ko_suffix). 

24# These are removed (not patched) so they are re-translated cleanly; the source 

25# fix changes the translation key, so the bad entries are never re-created. 

26# 

27# Precision: the mutated suffix is always an APPENDED short fragment, so we 

28# require a boundary (sentence period / dash / ellipsis / close-paren) followed 

29# by a short (<=24 char) lead-in before "관련 {noun}". A standalone legitimate 

30# description like "신규 상장 관련 공지" (the whole value, no preceding boundary) 

31# is therefore preserved — worst case for any residual false positive is a 

32# benign cache eviction → re-translation. 

33_MUTATION_ARTIFACT_RE = re.compile( 

34 r"[.。)\-–—…]\s*[가-힣A-Za-z0-9,()%·\s]{0,24}" 

35 r"관련\s?(?:통보|알림|공지|안내|보상|경고)(?:입니다)?\s*[.。]?\s*$" 

36) 

37 

38 

39def main() -> None: 

40 parser = argparse.ArgumentParser(description="Clean translation cache artifacts.") 

41 parser.add_argument("--dry-run", action="store_true", help="Preview changes without writing") 

42 args = parser.parse_args() 

43 

44 if not _CACHE_PATH.exists(): 

45 print("Translation cache not found, nothing to clean.") 

46 return 

47 

48 with open(_CACHE_PATH, encoding="utf-8") as f: 

49 cache = json.load(f) 

50 

51 fixed_count = 0 

52 removed_count = 0 

53 for key, value in list(cache.items()): 

54 # Remove suffix-mutation artifacts so they are re-translated cleanly. 

55 if _MUTATION_ARTIFACT_RE.search(value): 

56 removed_count += 1 

57 if args.dry_run: 

58 print(f" [remove {key[:8]}] {value[-60:]}") 

59 else: 

60 del cache[key] 

61 continue 

62 cleaned = _postprocess_translation(value) 

63 if cleaned != value: 

64 fixed_count += 1 

65 if args.dry_run: 

66 print(f" [{key[:8]}] {value[:60]}") 

67 print(f" → {cleaned[:60]}") 

68 else: 

69 cache[key] = cleaned 

70 

71 if fixed_count == 0 and removed_count == 0: 

72 print(f"Cache clean: {len(cache)} entries, 0 fixes needed.") 

73 return 

74 

75 if args.dry_run: 

76 print(f"\n[DRY RUN] Would fix {fixed_count} and remove {removed_count} of {len(cache)} entries.") 

77 else: 

78 with open(_CACHE_PATH, "w", encoding="utf-8") as f: 

79 json.dump(cache, f, ensure_ascii=False) 

80 print(f"Fixed {fixed_count}, removed {removed_count} of {len(cache)} cached translations.") 

81 

82 

83if __name__ == "__main__": 

84 main()
